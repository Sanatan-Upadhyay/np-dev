Q1
Consider the code snippet below and identify the correct statement/s for up-casting.

class Parent {
public:
  void sleep() {}
};

class Child : public Parent {
public:
  void gotoSchool() { }
};

int main() {
  Parent parent;
  Child child;
  return 0;
}
a)

Child *pChild = (Child *) & parent;

b)

Child *pChild = &parent;

c)

Parent *pParent = &child;

d)

Parent *pParent = (Parent *) &child;

Answer

c, d

}

int main() {
  Base b; // Line 2
  Base *bp = 0; // Line 3
  bp->show(); // Line 4
  return 0;

a)

Line 1: Pure virtual function in Base cannot have a body

Line 2: Cannot instantiate abstract class
